Introduction

This project uses an Arduino Uno, an IR Remote , an IR Receiver Module, a pushbutton , and a LED strip to turn on lights depending on what button is pressed on the ir remote and pushbutton.

Circuit Diagram

The IR receiver module recieves a signal from the IR Remote or if the pushbutton is pressed; the Arduino activates LEDs and the turns them on according to which button is pressed.

Environmental Impact Assessment

  • Materials: The Arduino Uno is RoHS and WEEE compliant, limiting hazardous substances.
  • Energy Efficiency: The circuit draws minimal power (~50 mA at 5V), ideal for USB or solar power.
  • Reusability: All components are modular and reusable, reducing electronic waste.
  • Packaging & Shipping: Ordering all components from a single supplier helps minimize carbon footprint.

Arduino Code

The following Arduino code controls the system, lighting LEDs and activating the buzzer based on distance readings from the HC-SR04 sensor.

  
  
#include 
#include   // Tinkercad uses the newer .hpp library

// Pin definitions
#define LED_PIN     6
#define IR_PIN      A0  // Change this if your IR receiver is on a different pin

// LED settings
#define NUM_LEDS    8
#define MAX_BRIGHTNESS 153  // 60% of 255

Adafruit_NeoPixel strip = Adafruit_NeoPixel(NUM_LEDS, LED_PIN, NEO_GRB + NEO_KHZ800);

// Pattern tracking
int currentPattern = 0; // 0=beam struggle, 1=all blue, 2=all purple
unsigned long lastAnimationUpdate = 0;
int animationStep = 0;

void setup() {
  Serial.begin(9600);
  
  // Initialize NeoPixels with 60% brightness
  strip.begin();
  strip.setBrightness(MAX_BRIGHTNESS);
  strip.show();
  
  // Initialize IR receiver with Tinkercad's library
  IrReceiver.begin(IR_PIN, ENABLE_LED_FEEDBACK);
  
  Serial.println("Dragon Ball Z LED Controller Ready!");
  Serial.println("Button 69: Beam Struggle, Button 70: All Blue, Button 71: All Purple");
}

void loop() {
  // Check IR remote for pattern changes
  if (IrReceiver.decode()) {
    handleIRCommand(IrReceiver.decodedIRData.command);
    IrReceiver.resume();
  }
  
  // If in beam struggle pattern, run animation
  if (currentPattern == 0) {
    runBeamStruggleAnimation();
  }
  
  delay(50);
}

void handleIRCommand(unsigned long command) {
  Serial.print("IR Command: ");
  Serial.println(command);
  
  switch(command) {
    case 69: // Button 1: Beam struggle animation
      currentPattern = 0;
      Serial.println("Pattern: Beam Struggle Animation");
      break;
    case 70: // Button 2: All blue
      currentPattern = 1;
      patternAllBlue();
      Serial.println("Pattern: All Blue");
      break;
    case 71: // Button 3: All purple
      currentPattern = 2;
      patternAllPurple();
      Serial.println("Pattern: All Purple");
      break;
    default:
      Serial.print("Unknown command: ");
      Serial.println(command);
      break;
  }
}

// PATTERN 1: Animated beam struggle
void runBeamStruggleAnimation() {
  if (millis() - lastAnimationUpdate > 200) { // Update every 200ms
    lastAnimationUpdate = millis();
    
    strip.clear();
    
    // Animate the beam struggle - beams push back and forth
    int beamMeetingPoint = 4 + sin(animationStep * 0.5) * 2; // Oscillates between ~2-6
    
    // Goku's beam (blue) from left
    for (int i = 0; i < beamMeetingPoint; i++) {
      strip.setPixelColor(i, 0, 0, 255);  // Blue
    }
    
    // Vegeta's beam (purple) from right
    for (int i = NUM_LEDS - 1; i >= beamMeetingPoint; i--) {
      strip.setPixelColor(i, 255, 0, 255);  // Purple
    }
    
    // Beam clash point (white with sparkle effect)
    strip.setPixelColor(beamMeetingPoint, 255, 255, 255);
    if (beamMeetingPoint > 0) strip.setPixelColor(beamMeetingPoint - 1, 128, 128, 255);
    if (beamMeetingPoint < NUM_LEDS - 1) strip.setPixelColor(beamMeetingPoint + 1, 255, 128, 255);
    
    strip.show();
    animationStep++;
  }
}

// PATTERN 2: All blue (static)
void patternAllBlue() {
  strip.clear();
  for (int i = 0; i < NUM_LEDS; i++) {
    strip.setPixelColor(i, 0, 0, 255);  // Blue
  }
  strip.show();
}

// PATTERN 3: All purple (static)
void patternAllPurple() {
  strip.clear();
  for (int i = 0; i < NUM_LEDS; i++) {
    strip.setPixelColor(i, 255, 0, 255);  // Purple
  }
  strip.show();
}

Conclusion

The system effectively demonstrates using infrared remote with an remote and a pushbutton to light up LEDs using Arduino. It’s affordable, educational, and sustainable through reusable components.
